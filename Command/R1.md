### Vyos R1 :

---------------

```bash
configure
set system host-name R1

set interfaces ethernet eth1 vif 1 description 'VLAN 1'
set interfaces ethernet eth1 vif 1 address '200.8.1.253/24'
set interfaces ethernet eth1 vif 1 address '2001:0:8:1::253/64'
set interfaces ethernet eth0 vif 10 description 'VLAN 10'
set interfaces ethernet eth0 vif 10 address '200.8.10.254/24'
set interfaces ethernet eth0 vif 10 address '2001:0:8:10::254/64'
set interfaces ethernet eth0 vif 20 description 'VLAN 20'
set interfaces ethernet eth0 vif 20 address '200.8.20.254/24'
set interfaces ethernet eth0 vif 20 address '2001:0:8:20::254/64'
set interfaces ethernet eth0 vif 99 description 'VLAN 99'
set interfaces ethernet eth0 vif 99 address '200.8.99.254/24'
set interfaces ethernet eth0 vif 99 address '2001:0:8:99::254/64'

# OSPFv2 :

configure
set interfaces loopback lo address 1.1.1.1/24
set protocols ospf parameters router-id 1.1.1.1
set protocols ospf area 0 network 200.8.1.0/24
set protocols ospf area 0 network 200.8.10.0/24
set protocols ospf area 0 network 200.8.20.0/24
set protocols ospf area 0 network 200.8.99.0/24

# OSPFv3 :

configure
set protocols ospfv3 area 0
set protocols ospfv3 paraleters router-id 1.1.1.1
--
set protocols ospfv3 interface eth0 area 0 #interface directement connecté à l'autre routeur
set protocols ospfv3 redistribute connected

# SLAAC + DHCP6-PD : 

set service router-advert interface eth1.10 prefix ::/64
set service router-advert interface eth1.20 prefix ::/64
set service router-advert interface eth1.10 name-server 2001:4860:4860::8844
set service router-advert interface eth1.20 name-server 2001:4860:4860::8844

# SSH :

set service ssh
set service ssh access-control allow user vyos

commit
save
exit

show ipv6 ospfv3 neighbor
show ipv6 ospfv3 neighbor detail
show ipv6 ospfv3 route
show ipv6 ospfv3 redistribute
show ip ospf route
show ip ospf neighbor
```
